	\section{Les règles de vol}
	\begin{frame}{Le vol à vue - VFR \anglais{Visual Flying Rules}}
	\begin{columns}
		\begin{column}{0.69\textwidth}
			\begin{itemize}
				\item Par beau temps (visibilité supérieure à 1500 m). \textbf{On parle de conditions météo de vol à vue}
				\item Le pilote utilise l'horizon naturel pour contrôler l'inclinaison de l'avion
				\item Le pilote utilise des repères au sol pour s'orienter
			\end{itemize}
			
			\astuce{Les pilotes francophones disent parfois que VFR signifie "Voies-ferrées/routes" car les voies ferrées et les routes constituent des bons repères de navigation}
		\end{column}
 		\begin{column}{0.31\textwidth}
 		\begin{figure}[H]
			\includegraphics[width=1\linewidth]{02-Navigation/img/vfr1.jpg}
			\legende{Vol en conditions VFR}{img:vfr1}
		
			\includegraphics[width=1\linewidth]{02-Navigation/img/vfr2.jpg}
			\legende{Vol en conditions VFR}{img:vfr2}
		\end{figure}
		\end{column}
	\end{columns}
		
	\end{frame}
	
	\begin{frame}{Le vol aux instruments - IFR \anglais{Instrument Flying Rules}}
	\begin{columns}
		\begin{column}{0.69\textwidth}
			\begin{itemize}
				\item Permet de voler par tous les temps
				\item Le pilote utilise des instruments pour gérer l'inclinaison de l'avion
				\item Le pilote utilise des moyens radios pour sa navigation
				\item Nécessite une qualification du pilote
				\item Nécessite un avion spécialement équipé
				\item Nécessite des équipements au sol/dans l'espace (GPS) pour le guidage en navigation et l'atterrissage
			\end{itemize}
		\end{column}
 		\begin{column}{0.31\textwidth}
 		\begin{figure}[H]
			\includegraphics[width=1.2\linewidth]{02-Navigation/img/ifr.jpg}
			\legende{Vol en conditions IFR}{img:ifr}
		\end{figure}
		\end{column}
	\end{columns}
	\end{frame}
