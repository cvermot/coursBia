 \section{L'atmosphère}
	\begin{frame}{L'atmosphère}
	
	\question{Qu'est-ce que l'atmosphère ?}
	\pause
	\question{De quoi est composée l'atmosphère terrestre ?}
	\end{frame}	 
	
	\begin{frame}{La composition de l'atmosphère terrestre}
	L'air sec est composé de :
	\begin{table}[H]
	\begin{tabular}{|l|c|}
		\hline
		Gaz & Proportion \\
		\hline
		\hline
		Azote ($N$) & 78~\% \\
		\hline
		Oxygène ($O$) & 21~\% \\
		\hline
		Argon ($Ar$) & 0,9~\% \\
		\hline
		Autres gaz (dont Dioxyde de carbone ($CO_2$)) & 0,1~\% \\
		\hline
	\end{tabular}
	\caption{Composition de l'atmosphère terrestre}
	\end{table}\pause
	
	L'air de l'atmosphère contient de la vapeur d'eau en quantité variable
	
	Rappel : la vapeur d'eau est un gaz incolore et inodore.
	\end{frame}

	\begin{frame}{Composition de l'atmosphère : influence de la quantité de vapeur d'eau}
		Lorsque la quantité de vapeur d'eau atteint 100\% de la quantité d'eau que peut contenir une masse d'air à une température donnée, l'eau en quantité excédentaire se condense : des micro gouttelettes se forment dans l'air. L'humidité devient apparente sous forme de nuages (en altitude) ou de brumes et brouillards (proches du sol).
		
		Pour une proportion d'eau fixée dans l'atmosphère, la température à laquelle la saturation est atteinte est appelée \textbf{point de rosée}.
	\end{frame}

	\begin{frame}{L'atmosphère standard}
			On fixe les caractéristiques d'une atmosphère dite "standard" (ISA) comme suit :
			\begin{itemize}
				\item Température au niveau de la mer : $\ang{15}C$
				\item Pression atmosphérique au niveau de la mer : $1013,25~hPa$
			\end{itemize}
			
			Dans cette atmosphère standard, ces grandeurs évoluent comme ceci :
			\begin{itemize}
				\item $\ang{-0.65}C$ tous les $100~m$ (soit $\ang{-2}C$ tous les $1000~ft$), jusqu'à la tropopause
				\item $-1~hPa$ tous les $8,5~m$ (soit $-1~hPa$ tous les $28~ft$)
			\end{itemize}
	\end{frame}

	\section{Les fronts}
	\begin{frame}{Les fronts}
	
		\begin{columns}
		\begin{column}{0.4\textwidth}
		Les masses d'air de l'atmosphère se différencient par leur température.
		
		La différence de température entre 2 masses d'air va provoquer, lors de leurs mouvements, le déplacement relatif sur le plan vertical des masses d'air.
		
		L'air chaud, plus léger, aura tendance à se soulever.
		\end{column}
		\begin{column}{0.6\textwidth}
		\begin{figure}[H]
			\includegraphics[width=1\linewidth]{03-Meteo/img/temsiEuroc20260127.png}
			\legende{Carte TEMSI}{img:temsiEuroc20260127}
		\end{figure}	
		\end{column}
		\end{columns}
	
	\end{frame}

	\begin{frame}{Front froid~~~~\includegraphics[width=0.3\linewidth]{03-Meteo/img/legendeFrontFroid.pdf}}
		\begin{figure}[H]
			\includegraphics[width=.5\linewidth]{03-Meteo/img/frontFroid.pdf}
		\end{figure}	
		
		L'air froid pousse l'air chaud, plus léger.
		
		A l'interface, l'humidité dans l'air chaud  se condense et forme des nuages développés verticalement : les précipitations sont brèves mais intenses.
		
			Après le passage du front froid, la traîne s'installe :  une zone de temps instable qui se forme à l'arrière d'un front froid. Elle est caractérisée par une visibilité variable et des averses intermittentes, intenses, voire des orages.
	\end{frame}

	\begin{frame}{Front chaud~~~~\includegraphics[width=0.3\linewidth]{03-Meteo/img/legendeFrontChaud.pdf}}
		\begin{figure}[H]
			\includegraphics[width=0.5\linewidth]{03-Meteo/img/frontChaud.pdf}
		\end{figure}	
		
		L'air chaud arrive sur l'air froid, plus dense. Il n'arrive pas à le pousser donc passe au-dessus. La vapeur d'eau contenue dans cet air chaud se condense au contact de l'air plus froid.
		
		Le début du front chaud est caractérisé par des nuages de haute altitude. Avec l'avancée du front, la base des nuages descend progressivement, on trouve des précipitations continues puis des nuages qui peuvent aller jusqu'au sol.
	\end{frame}
