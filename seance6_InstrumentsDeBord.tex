	\begin{frame}{Variomètre - fonctionnement}
		\begin{columns}
		\begin{column}{0.42\textwidth}
			\begin{itemize}
				\item Utilise la pression statique
				\item Une \textbf{fuite calibrée} dans le boitier permet de donner un retard entre la pression dans le boitier de l'instrument et la pression statique mesurée
				\item La mesure n'est pas instantanée (il y un délai entre la mise en montée/descente et l'affichage sur l'instrument).
			\end{itemize}
		\end{column}

	\subsection{Vitesse}
 	\begin{frame}{Anémomètre}
 		\begin{columns}
 		\begin{column}{0.40\textwidth}
		\begin{figure}	
		\centering
			\begin{tikzpicture}
				\badin{110}
			\end{tikzpicture}
		\legende{Un "badin" - graduation en n\oe uds (kt) - 1 kt = 1,852 km/h}{tikz:instrumentsBase}
		\end{figure}
		\end{column}
 		\begin{column}{0.60\textwidth}
 			Indique la vitesse
 			
 			Les arcs représentent différentes vitesses :
 			\begin{itemize}
 				\item arc blanc : vitesse avec volets sortis (VFE, Velocity Flaps Extended)
 				\item arc vert : vitesse en air turbulent (VNO, Velocity Normal Operating)
 				\item arc jaune : vitesse en air calme
 				\item trait rouge : vitesse à ne jamais dépasser (VNE, Velocity Never Exceeded)
 			\end{itemize}
 			\pause
 			\histoire{L'anémomètre qui équipe nos avions a été inventé en 1911 par l'ingénieur français Raoul Badin, qui a donc donné son nom à cet instrument.}
 		\end{column}
 		\end{columns}
 		
	\end{frame}	
	
	\begin{frame}{Anémomètre - fonctionnement}
		\begin{columns}
 		\begin{column}{0.58\textwidth}
		\begin{figure}
			% [trim={left bottom right top},clip]
			\includegraphics[width=1\linewidth]{01-EtudeAeronefs/img/instruments/schemaBadin.pdf}
			\legende{Fonctionnement d'un badin}{img:schemaBadin}

		\end{column}
		\begin{column}{0.42\textwidth}
		L'anémomètre utilise la \textbf{pression dynamique} en provenance du \textbf{tube Pitot}, et la \textbf{pression statique} en provenance de la \textbf{prise de pression statique}.
		\pause
		
		\histoire{Le tube Pitot a été inventé en 1732 par le physicien français Henri Pitot.}
		\end{column}
		\end{columns}
	\end{frame}

	\subsection{S'orienter par rapport au nord}
	\begin{frame}{Le compas magnétique}
 		\begin{columns}
 		\begin{column}{0.45\textwidth}
		\begin{figure}[H]	
		\centering
		\includegraphics[width=1\textwidth]{01-EtudeAeronefs/img/instruments/compasMagnetique.pdf}
		\legende{Un compas magnétique}{img:compasMagnetique}
		\end{figure}
		\end{column}
 		\begin{column}{0.55\textwidth}
 		
 		Le compas est une boussole, elle fonctionne grâce au magnétisme terrestre. Elle permet de lire le \textbf{cap magnétique}.
 		
 		Un liquide dans l'instrument permet de stabiliser un minimum, mais cet instrument reste relativement sensible aux vibrations et turbulences.
 			
 		\end{column}
 		\end{columns}
	\end{frame}
